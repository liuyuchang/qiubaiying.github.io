[题目 : I Think I Need a Houseboat ](http://poj.org/problem?id=1005)

# 直接模拟

-----

代码：

```
#include<cstdio>
using namespace std;
const int MAXN=2000;
int out[MAXN];
int main()
{
    int n;
    scanf("%d",&n);
    for(int j=1;j<=n;j++)
    {
        float x,y;
        scanf("%f%f",&x,&y);
        float rr=x*x+y*y;
        int i=int((3.1416/100.0)*rr)+1;
        out[j]=i;
    }
    for(int j=1;j<=n;j++)
    {
        printf("Property %d: This property will begin eroding in year %d.\n",j,out[j]);
    }
    printf("END OF OUTPUT.");
    return 0;
}
```
