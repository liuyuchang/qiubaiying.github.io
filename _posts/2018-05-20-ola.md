### 求欧拉函数的代码

``` 
int ola(int n)
{
    int ans=0,i,k;
    if(n==1)
    ans=1;
    else
    {
        ans=n;
        k=1;
        for(i=2;n!=1;i+=k)
        {
            if(n%i==0)
            {
                ans/=i;
                ans*=(i-1);
                while(n%i==0) n/=i;
                i=k;
            }
        }
    }
    return ans;
}
```
